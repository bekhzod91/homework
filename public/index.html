Documentation for API

# Create user

Request

curl -d '{"Email":"admin@example.com", "FirstName":"Admin", "SecondName": "Admin"}' -H "Content-Type: application/json" -X POST http://localhost:8080/api/v1/user/

Response

{
    "Data": {
        "Id": 1,
        "Email": "admin@example.com",
        "FirstName": "Admin",
        "SecondName": "Admin"
    }
}

# Get all user list

Request

curl http://localhost:8080/api/v1/user/

Response

{
        "Count": 1,
        "Data": [
            {
                "Id": 1,
                "Email": "admin@example.com",
            },
            {
                "Id": 2,
                "Email": "operator@example.com",
            }
        ]
}

# Get user detail

Response

{
    "Data": {
        "Id": 1,
        "Email": "admin@example.com",
        "FirstName": "Admin",
        "SecondName": "Admin"
    }
}

# Update user

Request

curl -d '{"Email":"admin1@example.com", "FirstName":"Admin1", "SecondName": "Admin1"}' -H "Content-Type: application/json" -X PUT http://localhost:8080/api/v1/user/1/

Response

{
    "Data": {
        "Id": 1,
        "Email": "admin1@example.com",
        "FirstName": "Admin1",
        "SecondName": "Admin1"
    }
}

# Delete user

Request

curl -X DELETE http://localhost:8080/api/v1/user/1/
